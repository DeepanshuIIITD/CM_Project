expDesign <- data.frame(row.names = colnames(allcount), condition = condition)

dds <- DESeqDataSetFromMatrix(countData = allcount, colData = expDesign, design = ~condition)

deseq2ResDF <- as.data.frame(res)

deseq2ResDF$significant <- ifelse(deseq2ResDF$padj < 0.1, "Significant", NA)

pos <- which(deseq2ResDF$significant == "Significant")
differential_genes <- deseq2ResDF[pos,]

up_deseq <- differential_genes[which(differential_genes$log2FoldChange > 0.54),]; up_genes_deseq <- rownames(up_deseq)

down_deseq <- differential_genes[which(differential_genes$log2FoldChange < -0.54),]; down_genes_deseq <- rownames(down_deseq)


dbs <- c("GO_Molecular_Function_2018", "GO_Cellular_Component_2018", "GO_Biological_Process_2018")
